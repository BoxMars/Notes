    \begin{algorithm}[H]
        \caption{Sum of three(A, K)}
        let $n \leftarrow |A|$ and assume $A = \{a_1, a_2, . . . , a_n\}$.

        \For(){$i==1,2,...,n-2$}{
            $j\leftarrow i+1$ and $k=n$.

            \While(){$k>j$}{
                \uIf{$a_i+a_j+a_k=K$}{
                    Output: $(i,j,k)$
                }
                \uElseIf{$a_i+a_j+a_k<K$}{
                    $j\leftarrow j+1$
                }
                \uElseIf{$a_i+a_j+a_k>K$}{
                    $k\leftarrow k-1$
                }
            }
        }
        Output: do not exist
    \end{algorithm}
